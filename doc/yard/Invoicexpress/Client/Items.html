def create_item(item, options={})
  raise(ArgumentError, "item has the wrong type") unless item.is_a?(Invoicexpress::Models::Item)

  params = { :klass => Invoicexpress::Models::Item, :body => item }
  post("items.xml", params.merge(options))
end
